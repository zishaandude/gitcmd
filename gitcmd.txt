# Initialize git repository
git init
# Add files to staging area
git add .
# Commit files
git commit -m "Initial commit"
# Rename branch to main
git branch -M main
# Connect local repo to GitHub repo (replace username if needed)
git remote add origin https://github.com/zishaandude/terraproject1.git
# Pull from GitHub first (important to avoid errors like "fetch first")
git pull origin main --rebase
# Push code to GitHub
git push -u origin main
# If you face conflicts or want to overwrite GitHub (use carefully)
# git push -u origin main --force
